Embed Script

Copy and paste this into your website just before </body>:

<script src="http://localhost:5001/static/chatbot-widget.js"></script>
<script>
  ChatbotWidget.updateConfig({
    apiUrl: 'http://localhost:5001',
    username: 'user',
    primaryColor: '#4f46e5',
    secondaryColor: '#7c3aed'
  });
</script>
